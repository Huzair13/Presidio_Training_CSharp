class Solution {
    public boolean isPalindrome(int x) {
        String str= String.valueOf(x);
        int size = str.length();
        String str2="";
        for(int i=size-1;i>=0;i--){
            str2+=str.charAt(i);
        }
        if(str.equals(str2)){
            return true;
        }
        return false;
    }
}
